export function t(lang: Lang, key: TKey, params?: Record<string, string | number>): string {
  let texto = STRINGS[lang][key]
  if (params) {
    for (const [k, v] of Object.entries(params)) {
      texto = texto.replace(`{${k}}`, String(v))
    }
  }
  return texto
}
